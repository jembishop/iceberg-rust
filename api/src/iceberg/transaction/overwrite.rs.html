35/// OverwriteAction is a transaction action for overwriting data files in the table.
36///
37/// Creates a snapshot with `Operation::Overwrite` semantics — adds new data files and
38/// optionally removes existing data files by rewriting affected manifests with those
39/// entries marked as `ManifestStatus::Deleted`.
40pub struct OverwriteAction {
41    check_duplicate: bool,
42    commit_uuid: Option<Uuid>,
43    key_metadata: Option<Vec<u8>>,
44    snapshot_properties: HashMap<String, String>,
45    added_data_files: Vec<DataFile>,
46    deleted_data_files: Vec<DataFile>,
47}
48
49impl OverwriteAction {
50    pub(crate) fn new() -> Self {
51        Self {
52            check_duplicate: true,
53            commit_uuid: None,
54            key_metadata: None,
55            snapshot_properties: HashMap::default(),
56            added_data_files: vec![],
57            deleted_data_files: vec![],
58        }
59    }
60
61    /// Set whether to check duplicate files.
62    pub fn with_check_duplicate(mut self, v: bool) -> Self {
63        self.check_duplicate = v;
64        self
65    }
66
67    /// Add data files to the snapshot.
68    pub fn add_data_files(mut self, data_files: impl IntoIterator<Item = DataFile>) -> Self {
69        self.added_data_files.extend(data_files);
70        self
71    }
72
73    /// Specify data files to be removed from the table in this overwrite.
74    pub fn delete_data_files(mut self, data_files: impl IntoIterator<Item = DataFile>) -> Self {
75        self.deleted_data_files.extend(data_files);
76        self
77    }
78
79    /// Set commit UUID for the snapshot.
80    pub fn set_commit_uuid(mut self, commit_uuid: Uuid) -> Self {
81        self.commit_uuid = Some(commit_uuid);
82        self
83    }
84
85    /// Set key metadata for manifest files.
86    pub fn set_key_metadata(mut self, key_metadata: Vec<u8>) -> Self {
87        self.key_metadata = Some(key_metadata);
88        self
89    }
90
91    /// Set snapshot summary properties.
92    pub fn set_snapshot_properties(mut self, snapshot_properties: HashMap<String, String>) -> Self {
93        self.snapshot_properties = snapshot_properties;
94        self
95    }
96}

98#[async_trait]
99impl TransactionAction for OverwriteAction {
100    async fn commit(self: Arc<Self>, table: &Table) -> Result<ActionCommit> {
101        let snapshot_producer = SnapshotProducer::new(
102            table,
103            self.commit_uuid.unwrap_or_else(Uuid::now_v7),
104            self.key_metadata.clone(),
105            self.snapshot_properties.clone(),
106            self.added_data_files.clone(),
107            self.deleted_data_files.clone(),
108        );
109
110        snapshot_producer.validate_added_data_files()?;
111
112        if self.check_duplicate {
113            snapshot_producer.validate_duplicate_files().await?;
114        }
115
116        let deleted_file_paths: HashSet<String> = self
117            .deleted_data_files
118            .iter()
119            .map(|f| f.file_path.clone())
120            .collect();
121
122        let snapshot_id = snapshot_producer.snapshot_id();
123        snapshot_producer
124            .commit(
125                OverwriteOperation {
126                    deleted_file_paths,
127                    snapshot_id,
128                },
129                DefaultManifestProcess,
130            )
131            .await
132    }
133}
134
135struct OverwriteOperation {
136    deleted_file_paths: HashSet<String>,
137    snapshot_id: i64,
138}
139
140impl SnapshotProduceOperation for OverwriteOperation {
141    fn operation(&self) -> Operation {
142        Operation::Overwrite
143    }
144
145    async fn delete_entries(
146        &self,
147        _snapshot_produce: &SnapshotProducer<'_>,
148    ) -> Result<Vec<ManifestEntry>> {
149        Ok(vec![])
150    }
151
152    async fn existing_manifest(
153        &self,
154        snapshot_produce: &SnapshotProducer<'_>,
155    ) -> Result<Vec<ManifestFile>> {
156        let Some(snapshot) = snapshot_produce.table.metadata().current_snapshot() else {
157            return Ok(vec![]);
158        };
159
160        let manifest_list = snapshot
161            .load_manifest_list(
162                snapshot_produce.table.file_io(),
163                &snapshot_produce.table.metadata_ref(),
164            )
165            .await?;
166
167        if self.deleted_file_paths.is_empty() {
168            return Ok(manifest_list
169                .entries()
170                .iter()
171                .filter(|entry| entry.has_added_files() || entry.has_existing_files())
172                .cloned()
173                .collect());
174        }
175
176        let mut result = Vec::new();
177
178        for manifest_file in manifest_list.entries() {
179            if !manifest_file.has_added_files() && !manifest_file.has_existing_files() {
180                continue;
181            }
182
183            let manifest = manifest_file
184                .load_manifest(snapshot_produce.table.file_io())
185                .await?;
186
187            let has_deletes = manifest.entries().iter().any(|entry| {
188                entry.is_alive() && self.deleted_file_paths.contains(entry.file_path())
189            });
190
191            if has_deletes {
192                let rewritten = self
193                    .rewrite_manifest(snapshot_produce, manifest_file, &manifest)
194                    .await?;
195                result.push(rewritten);
196            } else {
197                result.push(manifest_file.clone());
198            }
199        }
200
201        Ok(result)
202    }
203}
204
205impl OverwriteOperation {
206    /// Rewrite a manifest, marking entries whose file paths are in `deleted_file_paths`
207    /// as `ManifestStatus::Deleted`.
208    async fn rewrite_manifest(
209        &self,
210        snapshot_produce: &SnapshotProducer<'_>,
211        manifest_file: &ManifestFile,
212        manifest: &crate::spec::Manifest,
213    ) -> Result<ManifestFile> {
214        let table = snapshot_produce.table;
215
216        let new_manifest_path = format!(
217            "{}/metadata/{}-m-overwrite.avro",
218            table.metadata().location(),
219            Uuid::now_v7(),
220        );
221        let output_file = table.file_io().new_output(&new_manifest_path)?;
222        let builder = ManifestWriterBuilder::new(
223            output_file,
224            Some(self.snapshot_id),
225            manifest_file.key_metadata.clone(),
226            table.metadata().current_schema().clone(),
227            table.metadata().default_partition_spec().as_ref().clone(),
228        );
229
230        let mut writer = match table.metadata().format_version() {
231            FormatVersion::V1 => builder.build_v1(),
232            FormatVersion::V2 => match manifest_file.content {
233                ManifestContentType::Data => builder.build_v2_data(),
234                ManifestContentType::Deletes => builder.build_v2_deletes(),
235            },
236            FormatVersion::V3 => match manifest_file.content {
237                ManifestContentType::Data => builder.build_v3_data(),
238                ManifestContentType::Deletes => builder.build_v3_deletes(),
239            },
240        };
241
242        for entry in manifest.entries() {
243            if entry.is_alive() && self.deleted_file_paths.contains(entry.file_path()) {
244                let mut deleted: ManifestEntry = (**entry).clone();
245                deleted.snapshot_id = Some(self.snapshot_id);
246                writer.add_deleted_entry(deleted)?;
247            } else {
248                let cloned: ManifestEntry = (**entry).clone();
249                writer.add_existing_entry(cloned)?;
250            }
251        }
252
253        writer.write_manifest_file().await
254    }
255}

352            "test/3.parquet",

358        let updates = action_commit.take_updates();
359        let requirements = action_commit.take_requirements();
360
361        assert!(

413    #[tokio::test]
414    async fn test_overwrite_with_deleted_files() {
415        use crate::memory::tests::new_memory_catalog;
416        use crate::transaction::ApplyTransactionAction;
417        use crate::transaction::tests::make_v3_minimal_table_in_catalog;
418
419        let catalog = new_memory_catalog().await;
420        let table = make_v3_minimal_table_in_catalog(&catalog).await;
421        let spec_id = table.metadata().default_partition_spec_id();
422
423        let original_file = test_data_file("test/original.parquet", spec_id);
424        let tx = Transaction::new(&table);
425        let action = tx.fast_append().add_data_files(vec![original_file.clone()]);
426        let tx = action.apply(tx).unwrap();
427        let table = tx.commit(&catalog).await.unwrap();
428
429        let snapshot = table.metadata().current_snapshot().unwrap();
430        let manifest_list = snapshot
431            .load_manifest_list(table.file_io(), table.metadata())
432            .await
433            .unwrap();
434        assert_eq!(1, manifest_list.entries().len());
435
436        let replacement_file = test_data_file("test/replacement.parquet", spec_id);
437        let tx = Transaction::new(&table);
438        let action = tx
439            .overwrite()
440            .add_data_files(vec![replacement_file.clone()])
441            .delete_data_files(vec![original_file.clone()]);
442        let tx = action.apply(tx).unwrap();
443        let table = tx.commit(&catalog).await.unwrap();
444
445        let snapshot = table.metadata().current_snapshot().unwrap();
446        assert_eq!(snapshot.summary().operation, Operation::Overwrite);
447
448        let manifest_list = snapshot
449            .load_manifest_list(table.file_io(), table.metadata())
450            .await
451            .unwrap();
452
453        assert_eq!(2, manifest_list.entries().len());
454
455        let mut all_entries = vec![];
456        for manifest_file in manifest_list.entries() {
457            let manifest = manifest_file.load_manifest(table.file_io()).await.unwrap();
458            for entry in manifest.entries() {
459                all_entries.push((entry.status(), entry.file_path().to_string()));
460            }
461        }
462
463        assert!(
464            all_entries
465                .iter()
466                .any(|(status, path)| *status == ManifestStatus::Deleted
467                    && path == "test/original.parquet"),
468            "Original file should be marked as Deleted, entries: {all_entries:?}",
469        );
470
471        assert!(
472            all_entries
473                .iter()
474                .any(|(status, path)| *status == ManifestStatus::Added
475                    && path == "test/replacement.parquet"),
476            "Replacement file should be marked as Added, entries: {all_entries:?}",
477        );
478
479        // Verify snapshot summary reports the deleted file.
480        assert_eq!(
481            snapshot
482                .summary()
483                .additional_properties
484                .get("deleted-data-files")
485                .map(|s| s.as_str()),
486            Some("1")
487        );
488        assert_eq!(
489            snapshot
490                .summary()
491                .additional_properties
492                .get("deleted-records")
493                .map(|s| s.as_str()),
494            Some("1")
495        );
496
497        // Step 3: Fast append after overwrite — delete-only manifest must survive.
498        let appended_file = test_data_file("test/appended.parquet", spec_id);
499        let tx = Transaction::new(&table);
500        let action = tx.fast_append().add_data_files(vec![appended_file.clone()]);
501        let tx = action.apply(tx).unwrap();
502        let table = tx.commit(&catalog).await.unwrap();
503
504        let snapshot = table.metadata().current_snapshot().unwrap();
505        let manifest_list = snapshot
506            .load_manifest_list(table.file_io(), table.metadata())
507            .await
508            .unwrap();
509
510        // 3 manifests: rewritten (deleted entry), overwrite added, fast_append added.
511        assert_eq!(3, manifest_list.entries().len());
512
513        let mut all_entries = vec![];
514        for manifest_file in manifest_list.entries() {
515            let manifest = manifest_file.load_manifest(table.file_io()).await.unwrap();
516            for entry in manifest.entries() {
517                all_entries.push((entry.status(), entry.file_path().to_string()));
518            }
519        }
520
521        // The deleted entry must still be present after fast_append.
522        assert!(
523            all_entries
524                .iter()
525                .any(|(status, path)| *status == ManifestStatus::Deleted
526                    && path == "test/original.parquet"),
527            "Deleted entry should survive fast_append, entries: {all_entries:?}",
528        );
529    }
